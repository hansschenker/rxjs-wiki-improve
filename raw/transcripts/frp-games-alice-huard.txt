import { useState, useEffect } from "react";

type DirectionKeys = [boolean, boolean, boolean, boolean];

function useDirectionKeys(window: Window) {
  const [directionKey, setDirectionKey] = useState<DirectionKeys>([false, false, false, false]);

  useEffect(() => {
    function keyIsPressed(key: string): boolean {
      return window.document.activeElement !== null && window.document.activeElement === window.document.body && window.event?.key === key;
    }

    function updateDirectionKeys() {
      const l = keyIsPressed("ArrowLeft");
      const r = keyIsPressed("ArrowRight");
      const u = keyIsPressed("ArrowUp");
      const d = keyIsPressed("ArrowDown");
      setDirectionKey([l, r, u, d]);
    }

    window.addEventListener("keydown", updateDirectionKeys);
    window.addEventListener("keyup", updateDirectionKeys);

    return () => {
      window.removeEventListener("keydown", updateDirectionKeys);
      window.removeEventListener("keyup", updateDirectionKeys);
    };
  }, [window]);

  return directionKey;
}

function stateful<T>(initialValue: T, updateFn: (prev: T) => T): [T, () => void] {
  const [state, setState] = useState(initialValue);
  const updateState = () => setState(prev => updateFn(prev));
  return [state, updateState];

/**
 * Types and interfaces required for the translation.
 */

// Signal type representing a value that varies over time (Behavior).
export class Signal<A> {
  constructor(public sample: (time: number) => A) {}

  // Functor implementation for Signal (<$>)
  map<B>(f: (a: A) => B): Signal<B> {
    return new Signal((t) => f(this.sample(t)));
  }

  // Applicative implementation for Signal (<*>)
  apply<B>(f: Signal<(a: A) => B>): Signal<B> {
    return new Signal((t) => f.sample(t)(this.sample(t)));
  }

  static of<A>(value: A): Signal<A> {
    return new Signal(() => value);
  }
}

// SignalGen type representing a generator of signals (Now monad).
export class SignalGen<A> {
  constructor(public run: () => A) {}

  // Monad implementation for SignalGen (bind / >>=)
  flatMap<B>(f: (a: A) => SignalGen<B>): SignalGen<B> {
    return new SignalGen(() => f(this.run()).run());
  }

  static of<A>(value: A): SignalGen<A> {
    return new SignalGen(() => value);
  }
}

// Support types for the game domain
export type LevelNumber = number;
export type Score = number;
export type Health = number;

/**
 * generator :: Signal (SignalGen a) -> SignalGen (Signal a)
 */
export const generator = <A>(
  signal: Signal<SignalGen<A>>
): SignalGen<Signal<A>> => {
  return new SignalGen(() => new Signal((t) => signal.sample(t).run()));
};

/**
 * switcher helper (Internal implementation to support the usage below)
 */
export const switcher = <A>(
  signal: Signal<SignalGen<A>>
): SignalGen<A> => {
  return new SignalGen(() => signal.sample(0).run());
};

export async function playGame(
  directionkey: Signal<[boolean, boolean, boolean, boolean]>,
  levelCount_prime: Signal<LevelNumber>,
  score_prime: Signal<Score>,
  lives_prime: Signal<Health>
) {
  // - tm playGame matn function
  // (gameState, levelTrigger) <-
  // switcher $ playlevel directionkey <$> levelCount' <* > score' <* > lives'
  const [gameState, levelTrigger] = switcher(
    lives_prime.apply(
      score_prime.apply(
        levelCount_prime.map((l) => (s: Score) => (h: Health) =>
          playLevel(directionkey, l, s, h)
        )
      )
    )
  ).run();

  return { gameState, levelTrigger };
}

1.  **Feature Adaptation (Signal/SignalGen)**: The Haskell code uses `Signal` and `SignalGen` (typically from the `frp-now` library). In TypeScript, these are implemented as classes. `Signal` represents a "Behavior" (value over time) with `map` and `apply` methods to mimic Functor and Applicative behavior. `SignalGen` represents the "Now" monad for signal creation and state management.
2.  **Type System Differences**:
    -   Haskell's `(Bool, Bool, Bool, Bool)` is translated to the tuple type `[boolean, boolean, boolean, boolean]`.
    -   Type aliases `LevelNumber`, `Score`, and `Health` are used to maintain the semantic clarity of the original code.
    -   Haskell's `(Signal GameState, Signal Bool)` return type is translated to `[Signal<GameState>, Signal<boolean>]`.
3.  **Applicative Translation**: The Haskell applicative chain `playlevel directionkey <$> levelCount' <*> score' <*> lives'` is translated using nested `map` and `apply` calls. This preserves the functional dependency where `playLevel` is called with values sampled from the signals.
4.  **Apostrophes in Identifiers**: Variable names like `levelCount'` are not valid in TypeScript; they have been renamed to `levelCount_prime` to maintain distinctness while staying close to the original naming.
5.  **Signature Labels**: The line `event signats` in the Haskell signature was treated as a descriptive label/comment as it appeared between type definitions without a functional arrow, and was not represented as an argument in the actual call.
6.  **Monadic Binding**: The `<-` binding (destructuring in a monad) is translated to a destructuring assignment after executing the `SignalGen` via its `run()` method.
7.  **Preservation of Typos**: Comments like `- tm playGame matn function` were preserved exactly as they appeared in the source.
8.  **Switcher Implementation**: While `generator` was explicitly defined in the signature, `switcher` was used in the code. I provided an implementation of `switcher` that flattens the `Signal (SignalGen a)` structure into a `SignalGen a`, which is the behavior required by the destructuring assignment.
