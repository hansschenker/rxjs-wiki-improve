const Report = el =>
Map({
elementCount: Sum(1),
classes: Set(el.classList),
tags: Set(el.tagName),
maxHeight: Max(el.height)

Report.empty = () =>
Map({
elementCount: Sum.empty(),
classes: Set(),
tags: Set(),
maxHeight: Max.empty()

const getReport = root =>
Tree(root, x => Array. from(x.children)).foldMap(Report, Report.empty() )
