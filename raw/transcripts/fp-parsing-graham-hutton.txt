// Define the type for the result of a parse: a list of (value, remaining_input) pairs.
type ParseResult<T> = Array<[T, string]>;

// The Parser class wraps the parsing function.
// This is the equivalent of: newtype Parser a = P (String -> [(a, String)])
class Parser<T> {
  constructor(public readonly run: (input: string) => ParseResult<T>) {}
}

/**
 * The parse function executes a parser on an input string.
 * parse :: Parser a -> String -> [(a, String)]
 */
function parse<T>(p: Parser<T>, input: string): ParseResult<T> {
  return p.run(input);
}

/**
 * pure (or return) wraps a value into a parser that consumes no input.
 * pure :: a -> Parser a
 */
function pure<T>(val: T): Parser<T> {
  return new Parser(input => [[val, input]]);
}

/**
 * bind (>>=) chains parsers together.
 * (>>=) :: Parser a -> (a -> Parser b) -> Parser b
 */
function bind<T, U>(p: Parser<T>, f: (val: T) => Parser<U>): Parser<U> {
  return new Parser(input => {
    const results = p.run(input);
    // Flatten the results of applying the next parser to the remaining input of each result.
    return results.flatMap(([val, rest]) => f(val).run(rest));
  });
}

/**
 * map transforms the result of a parser.
 * fmap :: (a -> b) -> Parser a -> Parser b
 */
function map<T, U>(p: Parser<T>, f: (val: T) => U): Parser<U> {
  return bind(p, (val) => pure(f(val)));
}

/**
 * or (<|>) represents choice between two parsers.
 * (<|>) :: Parser a -> Parser a -> Parser a
 */
function or<T>(p: Parser<T>, q: Parser<T>): Parser<T> {
  return new Parser(input => {
    const res = p.run(input);
    // If the first parser fails (returns empty list), try the second parser.
    if (res.length === 0) return q.run(input);
    return res;
  });
}

/**
 * item is a primitive parser that consumes a single character.
 * item :: Parser Char
 */
const item: Parser<string> = new Parser(input => {
  if (input.length === 0) return [];
  return [[input[0], input.slice(1)]];

/**

function sat(p: (c: string) => boolean): Parser<string> {
  return bind(item, (c) => (p(c) ? pure(c) : new Parser(_ => [])));

  );
}

// Supporting types for the Parser Monad
export type Parser<T> = (input: string) => Array<[T, string]>;

export const Parser = {
    // Monadic return/pure for Parser
    return: <T>(x: T): Parser<T> => (input: string) => [[x, input]],
    
    // Monadic bind (>>=) for Parser
    bind: <T, U>(p: Parser<T>, f: (val: T) => Parser<U>): Parser<U> => 
        (input: string) => 
            p(input).reduce<[U, string][]>((acc, [x, nextInput]) => 
                acc.concat(f(x)(nextInput)), [])
};

// Expr Algebraic Data Type definition
export type Expr = 
  | { type: 'Val'; n: number }
  | { type: 'Dv'; ay: { x: Expr; y: Expr } };

// Maybe Monad implementation
export type Maybe<T> = Just<T> | Nothing;

export interface Just<T> {
  readonly tag: 'Just';
  readonly value: T;
}

export interface Nothing {
  readonly tag: 'Nothing';
}

export const just = <T>(value: T): Maybe<T> => ({ tag: 'Just', value });
export const nothing = (): Maybe<never> => ({ tag: 'Nothing' });

// return for the Maybe Monad
const return_ = just;

// bind (>>=) for the Maybe Monad to support monadic flows
const bind = <T, U>(m: Maybe<T>, f: (val: T) => Maybe<U>): Maybe<U> => {
    if (m.tag === 'Nothing') return nothing();
    return f(m.value);
};

// eval :: Expr -> Maybe Int
export const evalFunc = (expr: Expr): Maybe<number> => {
    // eval (Val n) = return n
    if (expr.type === 'Val') {
        const n = expr.n;
        return return_(n);
    }
    
    // eval (Dv ay) = do n <- eval x
    //                  m <- eval y
    if (expr.type === 'Dv') {
        const ay = expr.ay;
        const x = ay.x;
        const y = ay.y;
        
        // Emulating the monadic do-notation via nested bind calls
        return bind(evalFunc(x), (n) => 
            bind(evalFunc(y), (m) => {
                // Since the original Haskell snippet is a truncated sequence 
                // ending at the binding 'm <- eval y', we return pure(m) 
                // to represent the result of that final monadic operation.
                return return_(m);
            })
        );
    }
    
    // Exhaustive match safety (required in TypeScript)
    throw new Error("Pattern match failure");
};

/**
 * Required Dependencies: None.
 * This implementation uses standard TypeScript features.
 */

// --- Type Definitions for the Abstract Syntax Tree (AST) ---

/**
 * Represents the result of an expression parsing.
 * Matches grammar: expr ::= term + expr | term
 */
export type Expr = 
    | { type: 'Addition'; left: Term; right: Expr }
    | Term;

/**
 * Represents the result of a term parsing.
 * Matches grammar: term ::= factor * term | factor
 */
export type Term = 
    | { type: 'Multiplication'; left: Factor; right: Term }
    | Factor;

/**
 * Represents the result of a factor parsing.
 * Matches grammar: factor ::= (expr) | int
 */
export type Factor = 
    | { type: 'Grouping'; expression: Expr }
    | { type: 'Integer'; value: number };

// --- Lexer (Tokenizer) ---

/**
 * Token types required for the grammar.
 */
enum TokenType {
    PLUS = '+',
    STAR = '*',
    LPAREN = '(',
    RPAREN = ')',
    INT = 'INT',
    EOF = 'EOF'
}

/**
 * Token interface.
 */
interface Token {
    type: TokenType;
    value?: any;
}

/**
 * Lexer class to convert raw string input into a stream of tokens.
 */
class Lexer {
    private source: string;
    private tokens: Token[] = [];
    private start: number = 0;
    private current: number = 0;

    constructor(source: string) {
        this.source = source;
    }

    scanTokens(): Token[] {
        while (!this.isAtEnd()) {
            this.start = this.current;
            this.scanToken();
        }
        this.tokens.push({ type: TokenType.EOF });
        return this.tokens;
    }

    private scanToken() {
        const c = this.advance();
        switch (c) {
            case '(': this.addToken(TokenType.LPAREN); break;
            case ')': this.addToken(TokenType.RPAREN); break;
            case '+': this.addToken(TokenType.PLUS); break;
            case '*': this.addToken(TokenType.STAR); break;
            case ' ':
            case '\r':
            case '\t':
            case '\n':
                break;
            default:
                if (this.isDigit(c)) {
                    this.number();
                } else {
                    throw new Error(`Unexpected character: ${c}`);
                }
                break;
        }
    }

    private number() {
        while (this.isDigit(this.peek())) this.advance();
        const value = parseInt(this.source.substring(this.start, this.current));
        this.addToken(TokenType.INT, value);
    }

    private peek(): string {
        if (this.isAtEnd()) return '\0';
        return this.source.charAt(this.current);
    }

    private advance(): string {
        return this.source.charAt(this.current++);
    }

    private addToken(type: TokenType, value?: any) {
        this.tokens.push({ type, value });
    }

    private isDigit(c: string): boolean {
        return c >= '0' && c <= '9';
    }

    private isAtEnd(): boolean {
        return this.current >= this.source.length;
    }
}

// --- Parser ---

/**
 * Recursive Descent Parser implementing the grammar:
 * expr   ::= term + expr | term
 * term   ::= factor * term | factor
 * factor ::= (expr) | int
 */
export class Parser {
    private tokens: Token[];
    private current: number = 0;

    constructor(tokens: Token[]) {
        this.tokens = tokens;
    }

    /**
     * Parses an expression.
     * Logic: expr ::= term + expr | term
     */
    public expr(): Expr {
        const left = this.term();

        if (this.match(TokenType.PLUS)) {
            const right = this.expr();
            return { type: 'Addition', left, right };
        }

        return left;
    }

    /**
     * Parses a term.
     * Logic: term ::= factor * term | factor
     */
    private term(): Term {
        const left = this.factor();

        if (this.match(TokenType.STAR)) {
            const right = this.term();
            return { type: 'Multiplication', left, right };
        }

        return left;
    }

    /**
     * Parses a factor.
     * Logic: factor ::= (expr) | int
     */
    private factor(): Factor {
        if (this.match(TokenType.LPAREN)) {
            const expression = this.expr();
            this.consume(TokenType.RPAREN, "Expect ')' after expression.");
            return { type: 'Grouping', expression };
        }

        if (this.check(TokenType.INT)) {
            return { type: 'Integer', value: this.advance().value };
        }

        throw new Error(`Unexpected token: ${this.peek().type}. Expecting expression.`);
    }

    // --- Helper Methods ---

    private match(...types: TokenType[]): boolean {
        for (const type of types) {
            if (this.check(type)) {
                this.advance();
                return true;
            }
        }
        return false;
    }

    private consume(type: TokenType, message: string): Token {
        if (this.check(type)) return this.advance();
        throw new Error(message);
    }

    private check(type: TokenType): boolean {
        if (this.isAtEnd()) return false;
        return this.peek().type === type;
    }

    private advance(): Token {
        if (!this.isAtEnd()) this.current++;
        return this.previous();
    }

    private isAtEnd(): boolean {
        return this.peek().type === TokenType.EOF;
    }

    private peek(): Token {
        return this.tokens[this.current];
    }

    private previous(): Token {
        return this.tokens[this.current - 1];
    }
}

    const tokens = lexer.scanTokens();
    const parser = new Parser(tokens);
    return parser.expr();

/**
 * Result type for the Parser, representing either a successfully parsed value
 * and the remaining string, or null on failure.
 */
type ParseResult<T> = { value: T; rest: string } | null;

/**
 * A implementation of the Parser Monad to support Haskell's parser combinator logic.
 * This class provides the infrastructure for sequence (do notation) and choice (<|>).
 */
class Parser<T> {
    constructor(public parse: (input: string) => ParseResult<T>) {}

    /**
     * Equivalent to Haskell's (>>=) bind operator, used to chain parsers
     * in the translation of 'do' notation.
     */
    flatMap<U>(fn: (val: T) => Parser<U>): Parser<U> {
        return new Parser(input => {
            const res = this.parse(input);
            if (res === null) return null;
            return fn(res.value).parse(res.rest);
        });
    }

    /**
     * Equivalent to Haskell's (<|>) alternative operator.
     */
    or(other: Parser<T>): Parser<T> {
        return new Parser(input => {
            const result = this.parse(input);
            // If the first parser succeeds, return its result.
            // Otherwise, try the second parser with the original input.
            return result !== null ? result : other.parse(input);
        });
    }

    /**
     * Equivalent to Haskell's 'return' or 'pure' in the Monad/Applicative context.
     */
    static pure<T>(val: T): Parser<T> {
        return new Parser(input => ({ value: val, rest: input }));
    }
}

    /**

     * Lifts a value into the Parser context.

    static of<T>(value: T): Parser<T> {
        return new Parser(input => ({ value, rest: input }));
    }

     * Equivalent to Haskell's monadic bind (>>=).
     * Used to chain parsers together, simulating the 'do' block functionality.

    chain<U>(fn: (value: T) => Parser<U>): Parser<U> {
        return new Parser(input => {
            const result = this.parse(input);
            if (!result) return null;
            const nextParser = fn(result.value);
            return nextParser.parse(result.rest);
        });
    }

    /**
     * Equivalent to the Haskell alternative operator (<|>).
     * Attempts the first parser; if it fails, attempts the second.
     */
    or(other: Parser<T>): Parser<T> {
        return new Parser(input => {
            const result = this.parse(input);
            if (result !== null) return result;
            return other.parse(input);
        });
    }
}
