/**
 * Represents a discrete stream of events.
 */
export class Event<A> {
    private observers: ((value: A) => void)[] = [];

    /**
     * Subscribes a callback to the event stream.
     */
    public subscribe(observer: (value: A) => void): void {
        this.observers.push(observer);
    }

    /**
     * Triggers the event with a specific value.
     */
    public fire(value: A): void {
        this.observers.forEach((observer) => observer(value));
    }
}

/**
 * Represents a value that changes over time (a continuous signal).
 */
export class Behavior<B> {
    private value: B;
    private observers: ((value: B) => void)[] = [];

    constructor(initialValue: B) {
        this.value = initialValue;
    }

    /**
     * Samples the current value of the behavior.
     */
    public sample(): B {
        return this.value;
    }

    /**
     * Updates the behavior value and notifies subscribers.
     */
    public update(newValue: B): void {
        this.value = newValue;
        this.observers.forEach((observer) => observer(newValue));
    }

    /**
     * Subscribes to changes in the behavior.
     */
    public subscribe(observer: (value: B) => void): void {
        this.observers.push(observer);
        // Immediately notify with the current value upon subscription
        observer(this.value);
    }
}

/**
 * foldB :: (a -> b -> b) -> b -> Event a -> Behavior b
 * Accumulates events into a Behavior using a folding function and an initial value.
 */
export function foldB<A, B>(
    f: (val: A, acc: B) => B,
    initial: B,
    event: Event<A>
): Behavior<B> {
    const behavior = new Behavior<B>(initial);

    event.subscribe((a: A) => {
        const next = f(a, behavior.sample());
        behavior.update(next);
    });

    return behavior;
}

class Behavior<T> {
    private _value: T;
    private _listeners: ((val: T) => void)[] = [];

    constructor(initialValue: T) {
        this._value = initialValue;
    }

    /**
     * Returns the current value of the behavior.
     */
    get value(): T {
        return this._value;
    }

    /**
     * Updates the value and notifies subscribers.
     */
    set value(newValue: T) {
        this._value = newValue;
        this._listeners.forEach(handler => handler(newValue));
    }

    /**
     * Allows subscribing to changes in the behavior's value.
     */
    subscribe(handler: (val: T) => void): void {
        this._listeners.push(handler);
    }
}

/**

 * 
 * @param _ Ignored event value (unit/void in Haskell)
 * @param grid The current grid state
 * @returns The next grid state

function apply(_: void, grid: Grid): Grid {
    const nextGrid: Grid = grid.map(row => [...row]);
    
    for (let y = 0; y < GRID_SIZE; y++) {
        for (let x = 0; x < GRID_SIZE; x++) {
            let neighbors = 0;
            // Count neighbors in a Moore neighborhood
            for (let dy = -1; dy <= 1; dy++) {
                for (let dx = -1; dx <= 1; dx++) {
                    if (dx === 0 && dy === 0) continue;
                    
                    const ny = (y + dy + GRID_SIZE) % GRID_SIZE;
                    const nx = (x + dx + GRID_SIZE) % GRID_SIZE;
                    
                    if (grid[ny][nx]) {
                        neighbors++;
                    }
                }
            }

            // Apply Conway's Game of Life rules
            if (grid[y][x]) {
                // Survival
                nextGrid[y][x] = neighbors === 2 || neighbors === 3;
            } else {
                // Reproduction
                nextGrid[y][x] = neighbors === 3;
            }
        }
    }
    
    return nextGrid;
}
