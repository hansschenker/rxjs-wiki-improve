/**
 * PATTERN: Strategy Pipeline Builder
 * Automatic trace insertion at strategy boundaries
 * Makes complex pipelines learnable
 */
export class StrategyPipeline<T> {
  private strategies: Array<(source: Observable<T>) => Observable<T>> = [];
  
  add(name: string, strategy: (source: Observable<T>) => Observable<T>): this {
    this.strategies.push(
      (source) => source.pipe(
        trace(`→ ${name} START`),
        strategy as any,
        trace(`← ${name} END`)
      )
    );
    return this;
  }
  
  build(): (source: Observable<T>) => Observable<T> {
    return (source: Observable<T>) => {
      return this.strategies.reduce(
        (acc, strategy) => strategy(acc),
        source.pipe(trace('PIPELINE START'))
      ).pipe(
        trace('PIPELINE END')
      );
    };
  }
}

// Usage:
