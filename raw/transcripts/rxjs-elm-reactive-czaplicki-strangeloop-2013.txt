  const red = '#FF0000';

  const ngon = (

    />
  );

  // SVG circle with dashed red outline
  const circle = (

      cx="100"
      cy="100"
      r="70"
      fill="none"
      stroke={red}

function useKeyboardArrows() {
  const [position, setPosition] = useState<Point>({ x: 0, y: 0 });

  useEffect(() => {
    function handleKeyDown(event: KeyboardEvent) {
      const dir: Point = { x: 0, y: 0 };
      switch (event.key) {
        case 'ArrowUp':
          dir.y = -1;
          break;
        case 'ArrowDown':
          dir.y = 1;
          break;
        case 'ArrowLeft':
          dir.x = -1;
          break;
        case 'ArrowRight':
          dir.x = 1;
          break;
      }
      setPosition((pos) => ({ x: pos.x + dir.x, y: pos.y + dir.y }));
    }
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, []);

  return position;
}

export default function Main() {
  const pos = useKeyboardArrows();

  return (
    <div>
      {stack}
      {shapes(pos.x, pos.y)}
    </div>
  );
}
