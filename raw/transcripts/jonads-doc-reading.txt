import { fromEvent, timer, concat, Observable } from 'rxjs';
import { switchMap, mapTo } from 'rxjs/operators';

function counter(n: number, btn: HTMLElement): Observable<number> {
  return fromEvent(btn, 'click').pipe(
    switchMap(() =>
      concat(
        new Observable<number>(subscriber => {
          subscriber.next(n);
          subscriber.complete();
        }),
        counter(n + 1, btn)
      )
    )
  );
}

import { fromEvent, merge, timer } from 'rxjs';
import { filter, switchMap, map, takeUntil } from 'rxjs/operators';

function counter(n: number, btn: HTMLElement, win: Window) {
  const click$ = fromEvent(btn, 'click');
  const keydown$ = fromEvent<KeyboardEvent>(win, 'keydown');

  return merge(
    click$.pipe(
      switchMap(() =>
        timer(1000).pipe(
          map(() => n + 1),
          switchMap(newN => counter(newN, btn, win))
        )
      )
    ),
    keydown$.pipe(
      filter(event => event.key === 'Escape'),
      switchMap(() => counter(0, btn, win))
    )
  );
}

The choose operation takes a list of computations. Each of the computations in  the list carries (or produces) other computations. These wrapped computations represent 
the body of the clause that should be called when the clause is selected.
