/**
 * Utility function to generate an inclusive range of numbers,
 * mimicking Haskell's [start .. end] syntax.
 */
const range = (start: number, end: number): number[] =>
  end < start ? [] : Array.from({ length: end - start + 1 }, (_, i) => start + i);

/**
 * Utility function to perform a deep equality check on arrays,
 * used to replicate Haskell's list comparison (==).
 */
const arraysEqual = <T>(a: T[], b: T[]): boolean =>
  a.length === b.length && a.every((v, i) => v === b[i]);

/**
 * factors n =
 *   [x
 *   |x <- [1 .. n]
 *   , n mod x == 0
 *   ]
 */
function factors(n: number): number[] {
  return range(1, n)
    .filter(x =>
      n % x === 0
    );
}

/**
 * isprime n = factors n = [1,n]
 * 
 * Note: The second '=' is interpreted as an equality comparison (==),
 * which is the standard logic for a primality test based on factors.
 */
function isprime(n: number): boolean {
  return arraysEqual(factors(n), [1, n]);
}

/**
 * Variable alias to support the case-sensitive reference in the primes function.
 * Original code defines 'isprime' but calls 'isPrime'.
 */
const isPrime = isprime;

/**
 * primes n =
 *   [ p | p <- [2 .. n], isPrime p ]
 */
function primes(n: number): number[] {
  return range(2, n)
    .filter(p => isPrime(p));
}

2. **Range Implementation**: Haskell's `[1 .. n]` and `[2 .. n]` syntax represents inclusive ranges. Since TypeScript does not have a native range operator, a `range` helper function was implemented using `Array.from` to ensure identical behavior, including returning an empty array when the start exceeds the end.
3. **Array Equality**: In Haskell, the `==` operator compares the contents of lists. In TypeScript/JavaScript, `===` checks for reference equality for arrays. To maintain the exact functionality of `factors n == [1, n]`, I implemented an `arraysEqual` helper function to perform a value-based comparison.
4. **Modulo Operator**: The Haskell `mod` function (used as an infix operator in the source) was translated to the TypeScript `%` operator. For positive integers (the domain of these functions), both perform the same remainder calculation.
5. **Function Naming and Case Sensitivity**:
